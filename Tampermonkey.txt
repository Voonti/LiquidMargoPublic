(function() {
    'use strict';

    function appendCSS(url) {
        fetch(url)
            .then(response => response.text())
            .then(css => {
            var style = document.createElement("style");
            style.rel = "stylesheet";
            style.type = "text/css";
            style.innerHTML = css;
            document.head.appendChild(style);
        });
    }

    function appendJS(url) {
        fetch(url)
            .then(response => response.text())
            .then(jsCode => {
            eval(jsCode)
        })
            .catch(error => {
            console.error("Error fetching JavaScript:", error);
        })
    };

    function appendHTML(url) {
        fetch(url)
            .then(response => response.text())
            .then(html => {
            var baseElement = document.createElement('div');
            baseElement.innerHTML = html;
            document.documentElement.appendChild(baseElement);
        });
    }

    function attachClient() {
        const targetElement = document.querySelector("#GAME_CANVAS")
        if (targetElement) {
            appendHTML("https://raw.githubusercontent.com/Voonti/LiquidMargoPublic/develop/release.html");

        }
    }

    attachClient()
})();
